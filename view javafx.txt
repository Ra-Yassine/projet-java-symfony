    private void setupTable() {
        tableEnquetes.setColumnResizePolicy(TableView.CONSTRAINED_RESIZE_POLICY);
        tableEnquetes.setRowFactory(tv -> {
            TableRow<Enquete> row = new TableRow<>();
            row.setOnMouseClicked(event -> {
                if (event.getClickCount() == 2 && (!row.isEmpty())) {
                    Enquete enquete = row.getItem();
                    modifierEnquete(enquete);
                }
            });
            return row;
        });

        // Colonnes
        TableColumn<Enquete, Integer> colId = new TableColumn<>("ID");
        colId.setCellValueFactory(new PropertyValueFactory<>("idEnquete"));
        colId.setPrefWidth(50);

        TableColumn<Enquete, String> colSujet = new TableColumn<>("Sujet");
        colSujet.setCellValueFactory(new PropertyValueFactory<>("sujet"));
        colSujet.setPrefWidth(200);

        TableColumn<Enquete, String> colDescription = new TableColumn<>("Description");
        colDescription.setCellValueFactory(new PropertyValueFactory<>("description"));
        colDescription.setPrefWidth(250);

        TableColumn<Enquete, String> colStatut = new TableColumn<>("Statut");
        colStatut.setCellValueFactory(cellData -> 
            new SimpleStringProperty(cellData.getValue().getStatut().name()));
        colStatut.setCellFactory(column -> new TableCell<Enquete, String>() {
            @Override
            protected void updateItem(String item, boolean empty) {
                super.updateItem(item, empty);
                if (empty || item == null) {
                    setText(null);
                    setStyle("");
                } else {
                    setText(item);
                    if ("ACTIVE".equals(item)) {
                        setStyle("-fx-text-fill: green; -fx-font-weight: bold;");
                    } else {
                        setStyle("-fx-text-fill: red; -fx-font-weight: bold;");
                    }
                }
            }
        });
        colStatut.setPrefWidth(100);

        TableColumn<Enquete, String> colAnonyme = new TableColumn<>("Anonyme");
        colAnonyme.setCellValueFactory(cellData -> 
            new SimpleStringProperty(cellData.getValue().isAnonyme() ? "✓ Oui" : "✗ Non"));
        colAnonyme.setPrefWidth(80);

        TableColumn<Enquete, String> colDate = new TableColumn<>("Date création");
        colDate.setCellValueFactory(cellData -> {
            DateTimeFormatter formatter = DateTimeFormatter.ofPattern("dd/MM/yyyy HH:mm");
            return new SimpleStringProperty(cellData.getValue().getDateCreation().format(formatter));
        });
        colDate.setPrefWidth(150);

        TableColumn<Enquete, String> colQuestions = new TableColumn<>("Questions");
        colQuestions.setCellValueFactory(cellData -> {
            int nbQuestions = cellData.getValue().getAttributs() != null ? 
                             cellData.getValue().getAttributs().size() : 0;
            return new SimpleStringProperty(nbQuestions + " question(s)");
        });
        colQuestions.setPrefWidth(100);

        tableEnquetes.getColumns().addAll(colId, colSujet, colDescription, colStatut, 
                                          colAnonyme, colDate, colQuestions);
    }

    private void chargerDonnees() {
        try {
            List<Enquete> enquetes = service.getAll();
            enquetesList.clear();
            enquetesList.addAll(enquetes);
            tableEnquetes.setItems(enquetesList);
            updateStatus(enquetesList.size() + " enquête(s) chargée(s)");
        } catch (Exception e) {
            e.printStackTrace();
            showAlert("Erreur lors du chargement des données");
        }
    }

    private void rechercherEnquetes(String query) {
        try {
            List<Enquete> resultats = new ArrayList<>();
            for (Enquete e : service.getAll()) {
                if (e.getSujet().toLowerCase().contains(query) ||
                    e.getDescription().toLowerCase().contains(query)) {
                    resultats.add(e);
                }
            }
            enquetesList.clear();
            enquetesList.addAll(resultats);
            tableEnquetes.setItems(enquetesList);
            updateStatus(resultats.size() + " résultat(s) pour \"" + query + "\"");
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    private void modifierEnquete() {
        Enquete selected = tableEnquetes.getSelectionModel().getSelectedItem();
        if (selected == null) {
            showAlert("Veuillez sélectionner une enquête à modifier");
            return;
        }
        modifierEnquete(selected);
    }

    private void modifierEnquete(Enquete enquete) {
        // Créer une boîte de dialogue de modification
        Dialog<Enquete> dialog = new Dialog<>();
        dialog.setTitle("Modifier l'enquête");
        dialog.setHeaderText("Modification de : " + enquete.getSujet());

        ButtonType modifierButtonType = new ButtonType("Modifier", ButtonBar.ButtonData.OK_DONE);
        dialog.getDialogPane().getButtonTypes().addAll(modifierButtonType, ButtonType.CANCEL);

        GridPane grid = new GridPane();
        grid.setHgap(10);
        grid.setVgap(10);
        grid.setPadding(new Insets(20, 150, 10, 10));

        TextField sujetField = new TextField(enquete.getSujet());
        sujetField.setPrefWidth(300);
        TextArea descArea = new TextArea(enquete.getDescription());
        descArea.setPrefRowCount(3);
        descArea.setPrefWidth(300);
        CheckBox anonymeCheck = new CheckBox("Anonyme");
        anonymeCheck.setSelected(enquete.isAnonyme());

        ComboBox<StatutEnquete> statutCombo = new ComboBox<>();
        statutCombo.getItems().setAll(StatutEnquete.values());
        statutCombo.setValue(enquete.getStatut());

        grid.add(new Label("Sujet:"), 0, 0);
        grid.add(sujetField, 1, 0);
        grid.add(new Label("Description:"), 0, 1);
        grid.add(descArea, 1, 1);
        grid.add(new Label("Statut:"), 0, 2);
        grid.add(statutCombo, 1, 2);
        grid.add(anonymeCheck, 1, 3);

        dialog.getDialogPane().setContent(grid);

        dialog.setResultConverter(dialogButton -> {
            if (dialogButton == modifierButtonType) {
                enquete.setSujet(sujetField.getText());
                enquete.setDescription(descArea.getText());
                enquete.setStatut(statutCombo.getValue());
                enquete.setAnonyme(anonymeCheck.isSelected());
                return enquete;
            }
            return null;
        });

        Optional<Enquete> result = dialog.showAndWait();
        result.ifPresent(enqueteModifiee -> {
            try {
                service.modifier(enqueteModifiee);
                chargerDonnees();
                showInfo("Succès", "Enquête modifiée avec succès !");
            } catch (Exception e) {
                e.printStackTrace();
                showAlert("Erreur lors de la modification");
            }
        });
    }

    private void supprimerEnquete() {
        Enquete selected = tableEnquetes.getSelectionModel().getSelectedItem();
        if (selected == null) {
            showAlert("Veuillez sélectionner une enquête à supprimer");
            return;
        }

        Alert confirm = new Alert(Alert.AlertType.CONFIRMATION);
        confirm.setTitle("Confirmation de suppression");
        confirm.setHeaderText("Supprimer l'enquête");
        confirm.setContentText("Êtes-vous sûr de vouloir supprimer l'enquête : " + selected.getSujet() + " ?");

        Optional<ButtonType> result = confirm.showAndWait();
        if (result.isPresent() && result.get() == ButtonType.OK) {
            try {
                service.supprimer(selected.getIdEnquete());
                chargerDonnees();
                showInfo("Succès", "Enquête supprimée avec succès !");
            } catch (Exception e) {
                e.printStackTrace();
                showAlert("Erreur lors de la suppression");
            }
        }
    }

    private void changerStatut(boolean activer) {
        Enquete selected = tableEnquetes.getSelectionModel().getSelectedItem();
        if (selected == null) {
            showAlert("Veuillez sélectionner une enquête");
            return;
        }

        try {
            if (activer) {
                service.activer(selected.getIdEnquete());
                showInfo("Succès", "Enquête activée avec succès !");
            } else {
                service.desactiver(selected.getIdEnquete());
                showInfo("Succès", "Enquête désactivée avec succès !");
            }
            chargerDonnees();
        } catch (Exception e) {
            e.printStackTrace();
            showAlert("Erreur lors du changement de statut");
        }
    }

    private void voirDetailsEnquete() {
        Enquete selected = tableEnquetes.getSelectionModel().getSelectedItem();
        if (selected == null) {
            showAlert("Veuillez sélectionner une enquête");
            return;
        }

        StringBuilder details = new StringBuilder();
        details.append("ID: ").append(selected.getIdEnquete()).append("\n");
        details.append("Sujet: ").append(selected.getSujet()).append("\n");
        details.append("Description: ").append(selected.getDescription()).append("\n");
        details.append("Statut: ").append(selected.getStatut()).append("\n");
        details.append("Anonyme: ").append(selected.isAnonyme() ? "Oui" : "Non").append("\n");
        details.append("Date création: ").append(selected.getDateCreation()).append("\n");
        details.append("\nQuestions:\n");
        
        if (selected.getAttributs() != null && !selected.getAttributs().isEmpty()) {
            for (int i = 0; i < selected.getAttributs().size(); i++) {
                details.append(i+1).append(". ").append(selected.getAttributs().get(i).getNom()).append("\n");
            }
        } else {
            details.append("Aucune question");
        }

        TextArea textArea = new TextArea(details.toString());
        textArea.setEditable(false);
        textArea.setPrefRowCount(15);
        textArea.setPrefWidth(400);

        Dialog<Void> dialog = new Dialog<>();
        dialog.setTitle("Détails de l'enquête");
        dialog.getDialogPane().setContent(textArea);
        dialog.getDialogPane().getButtonTypes().add(ButtonType.CLOSE);
        dialog.showAndWait();
    }

    private void updateStatus(String message) {
        statusLabel.setText("📌 " + message + " - " + java.time.LocalDateTime.now().format(DateTimeFormatter.ofPattern("HH:mm:ss")));
    }

    private void showAlert(String message) {
        Alert alert = new Alert(Alert.AlertType.WARNING);
        alert.setHeaderText(null);
        alert.setContentText(message);
        alert.showAndWait();
    }

    private void showInfo(String title, String message) {
        Alert alert = new Alert(Alert.AlertType.INFORMATION);
        alert.setTitle(title);
        alert.setHeaderText(null);
        alert.setContentText(message);
        alert.showAndWait();
    }

    private void showMessage(String msg) {
        Alert alert = new Alert(Alert.AlertType.INFORMATION);
        alert.setHeaderText(null);
        alert.setContentText(msg);
        alert.showAndWait();
    }
